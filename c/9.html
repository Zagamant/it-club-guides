Описание и использование цикла for.

Вы уже изучили два цикла языка Си (while и do-while), есть ещё один цикл for

Общий вид цикла for

						
	for(начальная_инициализация; условие; операторы2)
	{
		операторы1;
	}

Начальная инициализация выполныется один раз перед началом работы цикла. В этом блоке обычно объявляется переменная и её присваивается начальное значение. Затем проверяется условие, если условие верно, то выполняются операторы1(операторы в теле цикла), затем выполняются операторы2 и снова проверяется условие и т.д. пока условие не нарушится. В качестве операторов2 обычно написано, как будет меняться переменная цикла.

Выведем числа от 1 до 10 циклом for

		
	for(int i=1; i<=10; i++)
	{
		printf("%d ",i);
	}

Переменная обьявленная в цикле for доступна только в пределах этого цикла. Можно использовать одну и туже переменную в нескольких циклах for.

		
	//внутренняя переменная счетчик
	for(int i=1; i<=10; i++) 
	{
		printf("%d ",i);
	}
	for(int i=-10; i>0; i-=2)
	{
		printf("%d ",i);
	}
	int sum=0;
	for(int i=1; i<=100; i++)
	{
		sum=sum+i;
	}
	printf("Сумма чисел от 1 до 100 = %d ",sum);

При таком использованиее место в памяти под переменную i выделяется при выполнении цикла for и, как только цикл замершается, место в памяти освобождается. Переменную для цикла можно объявить один раз в начале программы и использовать во всех циклах, в таком случае переменная будет доступна во всей программе и память освободиться только после окончания программы.

		
	//внешняя переменная счетчик
	int i; //объявление одной переменной для всех циклов
	for(i=1; i<=10; i++) 
	{
		printf("%d ",i);
	}
	for(i=-10; i>0; i-=2)
	{
		printf("%d ",i);
	}
	int sum=0;
	for(i=1; i<=100; i++)
	{
		sum=sum+i;
	}
	printf("Сумма чисел от 1 до 100 = %d ",sum);

Внутренняя переменная счетчик используется чаще всего!

Замена циклов while и do-while на цикл for.

Все циклы в языке Си взаимозаменяемы, т.е. всегда любой из циклов можно заменить на другой. Однако, для разных ситуаций удобно использовать разлицные циклы. Если известно, что переменная меняется в каком-то заданном промежутке, то используется цикл for. Если работа цикла зависит от какого-то начального условия, то используется цикл while. Если сначала необходимо выполнить какие-то действия, а только потом проверять условие, то исрользуется цикл do-while.

Все три цикла делают одно и тоже: выводят числа от 1 до 10.

while
(пока, оператор с предусловием)		 do-while
(делать-пока, оператор с постусловием)		 for
(для, оператор для заданных значений переменной)		 
						
#define _CRT_SECURE_NO_WARNINGS
#include <stdio.h>
#include <locale.h>
int main()
{
	int x=1;
	while(x <= 10)
	{
		printf("%d ",x);
		x++;
	}
	return 0;
}
						
		
						
#define _CRT_SECURE_NO_WARNINGS
#include <stdio.h>
#include <locale.h>
int main()
{
	int x=1;
	do
	{
		printf("%d ",x);
		x++;
	}while(x <= 10);
	return 0;
}
						
		
						
#define _CRT_SECURE_NO_WARNINGS
#include <stdio.h>
#include <locale.h>
int main()
{
	
	for(int x=1; x <= 10; x++)
	{
		printf("%d ",x);
	}

	return 0;
}

Циклы for с неполным описанием параметров.

В цикле for любая часть может отсутствовать, главное сохранить две ';'. Если отсутствует условие, то получаем бусконечный цикл, т.е. вместо условия всегда true.

						
	int x=1;
	for(; x <= 10; x++) //отсутствует начальная инициализация
	{
		...
	}

	int x=1;
	for(; x <= 10; ) //отсутствует начальная инициализация и операторы2, цикл for превратился в while
	{
		...
		x++;
	}

	for( ; ; )//отсутствует все - бесконечный цикл
	{
		...
	}

	for(int x=0; ; x++)//отсутствует условие - бесконечный цикл со счетчиком
	{
		...
	}

Операторы break, continue.

При работе с циклами используются операторы break и continue.

Оператор break используется для принудительной остановки цикла. (Можно использовать для любого цикла)

Оператор continue пропускает все операторы тела цилка после оператора continue и цикл продолжает свою работу. (Можно использовать для любого цикла)

Программа вычисляет сумму всех вводимых чисел пока не будет введен 0.

						
#define _CRT_SECURE_NO_WARNINGS
#include <stdio.h>
#include <locale.h>
int main()
{
	setlocale(LC_ALL,"Russian");
	int n, sum=0;
	
	for( ; ; )
	{	
		printf("введите чило: ");
		scanf("%d", &n);
		if(n == 0)
		{
			break; //принудительная остановка цикла
		}
		sum+=n;
	}
	printf("сумма всех введенных чисел = %d", sum);
	return 0;
}

Программа выводит числа от 1 до 10 пропуская 5.

						
#define _CRT_SECURE_NO_WARNINGS
#include <stdio.h>
#include <locale.h>
int main()
{
	setlocale(LC_ALL,"Russian");
	for(int i=1; i<=10; i++)
	{	
		if(i == 5)
		{
			continue; //пропускается одно выполнение цикла
		}
		printf("%d\t", i);
	}
	
	return 0;
}
