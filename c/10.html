Введение в массивы. Создание массивов.

Массив - это совокупность переменных одного типа, к которым обращаются с помощью общего имени. Доступ к отдельному элементу массива может осуществляться с помощью индекса. Нумерация элементов в массиве начинается с 0. В Си все массивы состоят из соприкасающихся участков памяти. Наименьший адрес соответствует первому элементу, наибольший - последнему элементу. Массивы могут иметь одну или несколько размерностей.

Создание массива

Стандартный вид объявления одномерного массива следующий:

						
тип имя_переменной [размер];

В Си массивы должны определяться однозначно, чтобы компилятор мог выделить для них место в памяти. Здесь тип объявляет базовый тип массива и является типом каждого элемента массива. Параметр размер определяет, сколько элементов содержит массив.

У всех массивов первый элемент имеет индекс 0. Поэтому, если написать

						
int mas [10];

то будет объявлен массив символов из 10 элементов, причем эти элементы адресуются индексом от 0 до 9.

В Си размер массива может быть только константой.

Константа - переменная значение которой задается один раз и никогда не меняется. (Например число PI в математике).

						
const int n = 10;
int mas[n];

Ввод, вывод массивов.

Следующая программа заполняет целочисленный массив числами от 0 до 9 и выводит его:

						
#include <stdio.h>
int main()
{
	const int n = 10; //размер массива хранится в константе
	int mas[n]; // резервирует место для 10 целочисленных элементов
	//заполнение массива 
	for (int i = 0; i < n; i++)
	{
		mas[i]=i;
	}
	//вывод массива в прямом порядке
	for (int i = 0; i < n; i++)
	{
		printf("%d\t", mas[i]);
	}
	printf("\n");
	//вывод массива в обратном порядке
	for (int i = n-1; i >= 0; i--)
	{
		printf("%d\t", mas[i]);
	}
}

Заполнение массива числами введенными с клавиатуры и вывод.

						
#include <stdio.h>

{

		scanf("%d", &mas[i]);
	}
	//вывод массива
	for (int i = 0; i < n; i++)
	{
		printf("%d\t", mas[i]);
	}
}

Решение частых задач с массивами - поиск элементов, обработка данных, изменение элементов массивов.

Рассмотрим несколько примеров по работе с массивами.

Ввести массив и вывести нечетные элементы массива.

						
#define _CRT_SECURE_NO_WARNINGS
#include <stdio.h>
int main()
{
	const int  n = 10;
	int array[n];
	for (int i = 0; i < n; i++) 
	{
		printf("array[%d] = ", i);
		scanf("%d", &array[i]);
	}
	for (int i = 0; i < n; i++)
	{
		if (array[i] % 2 == 1)
		{
			printf("array[%d] = %d\n", i, array[i]);
		}
	}
	return 0;
}

		}

  • Преобразовать массив так, чтобы сначала шли нулевые элементы, а затем все остальные.

    • 						
#define _CRT_SECURE_NO_WARNINGS
#include <stdio.h>
int main()
{
	const int  n = 10;
	int a[n];
	for (int i = 0; i < n; i++)
	{
		printf("a[%d] = ", i);
		scanf("%d", &a[i]);
	}
	for (int i = 0; i < n; i++)
	{
		printf("%d\t",  a[i]);
	}
	printf("\n");
	int i1 = 0, i2 = n - 1;
	while (i1 < i2)
	{
		while (i1 < i2 && !a[i1])
		{
			i1++;
		}
		while (i2 > i1 && a[i2])
		{
			i2--;
		}
		if (i1 < i2)
		{
			int tmp = a[i1];
			a[i1] = a[i2];
			a[i2] = tmp;
		}
		i1++;
		i2--;
	}
	for (int i = 0; i < n; i++)
	{
		printf("%d\t", a[i]);
	}	
	return 0;
}
