Двумерные и n-мерные массивы.

Двумерный массив - это массив состоящий из одномерных массивов. У каждого элемента двумерного массива есть две координаты - номер строчки(i) и номер столбца(j).

Создание массива

Стандартный вид объявления двумерного массива следующий:

						
тип имя_переменной [количество строк][количество столбцов];

При создании двумерного массива необходимо указать количество строк и количество столбцов.

Пример объявления двумерного массива: 

						
const int row = 10;
const int column = 5;
int mas[row][column];

Чтобы обратиться к одному элементу массива необходимо указать номер строчки и номер столбца. Нумерация, как и в одномерном массиве, с нуля.

Ввод, вывод двумерного массива.

Следующая программа заполняет целочисленный массив случайными числами и выводит его:

						
#define _CRT_SECURE_NO_WARNINGS
#include <stdio.h>
#include <stdlib.h>
#include <time.h>
int main()
{
	const int row = 10;   //количество строк в массиве
	const int column = 5; //количество столбцов в массиве
	int mas[row][column]; //создаем массив на 10 строк и 5 столбцов
	srand(time(NULL));
	//заполнение двумерного массива 
	for (int i = 0; i < row; i++) //цикл переберающий строки
	{
		for (int j = 0; j < column; j++) //цикл перебирающий столбцы
		{
			mas[i][j]=rand()%100; //заполнение случайными числами
		}
	}

	//вывод массива
	for (int i = 0; i < row; i++)
	{
		for (int j = 0; j < column; j++)
		{
			printf("%d\t", mas[i][j]); //вывод всех данных одной строки
		}
		printf("\n");//переход на новую строку
	}
}

Создание многомерных массивов.

Аналогично с двумерным можно создать трехмерный, четырехмерный и т. д. массив.

При объявлении массива указываем ещё одни [] и для перебора всех элементов используем ещё один цикл for.

Пример использования трёхмерного массива.

						
#define _CRT_SECURE_NO_WARNINGS
#include <stdio.h>
#include <stdlib.h>
#include <time.h>
int main()
{
	const int row = 10;   //количество строк в массиве
	const int column = 5; //количество столбцов в массиве
	const int layer = 3;  //количество слоев в массиве
	int mas[row][column]; //создаем массив на 10 строк, 5 столбцов и 3 слоя
	srand(time(NULL));
	//заполнение трехмерного массива 
	for (int i = 0; i < row; i++) //цикл переберающий строки
	{
		for (int j = 0; j < column; j++) //цикл перебирающий столбцы
		{
			for (int k = 0; k < layer; k++) //цикл перебирающий слои
			{
				mas[i][j][k]=rand()%100; //заполнение случайными числами
			}
		}
	}

	//вывод массива
	for (int i = 0; i < row; i++)
	{
		for (int j = 0; j < column; j++)
		{
			for (int k = 0; k < layer; k++)
			{
				printf("%d\t", mas[i][j][k]); //вывод всех данных одной строки
			}
		printf("\n");//переход на новую строку
		}
	printf("\n\n");//переход на новый слой
	}
}
