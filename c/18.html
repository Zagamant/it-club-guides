Функции с параметрами и без параметров.

Функция (подпрограмма) — фрагмент программного кода, к которому можно обратиться из другого места программы.

До этого весь код мы писали в функции main - это главная функция, с неё начинается выполнение кода. Кроме main мы можем писать любые свои функции.

Все наши функции должны быть написаны после подключения библиотек (include) и перед функцией main!

Шаблон для создания функции:

						
возвращаемый_тип имя_функции(передаваемые_в_функцию_переменные)
{
	тело функции, тот код, который выполняет данная функция
	return значение; //return писать не нужно, если функция ничего не возвращает
}

Разберемся с каждой частью описания функции.

Возвращаемый тип - это тип переменной, которая хранит результат работы функции. Если функция ничего не возвращает, то пишем void. Если после работы функции, результат нужно сохранить для дальнейшего использования, то указываем возвращаемый тип. Если функция что-то вычисляет и ответ сразу выводит, то возвращать ничего не нужно и пишем void. Желательно чтобы функция возвращала результат.

Имя функции - это название, которое мы сами придумываем для функции. Имя функции должно описывать, что будет делать данная функция. Имена функций в Си принято писать с маленькой буквы, если необходимо несколько слов, то следующее слово пишем с большой буквы, но без пробелов. Можно использовать в имени знак подчеркивание '_'. Например: createArray (создать массив), printArray (вывести массив), sort(сортировка), summ(сумма) и т.д.

Передаваемые в функцию переменные(параметры) - это данные, которые передаются в функцию при вызове, эти данные функция использует в процессе работы. Для каждой переменной необходима указывать тип (int a, int b, double c - верно, int a, b, c - не верно). Если функция работает без парметров, то ставим пустые круглые скобки ().

Тело функции - это логика работы функции, все действия, которые функция должна выполнить.

return значение - это возврат результата работы функции. Если функция ничего не возвращает, то return писать не нужно.

Рассмотрим несколько примеров.

Функция вычисляет сумму двух чисел:

						
#define _CRT_SECURE_NO_WARNINGS
#include <stdio.h>

//функция без параметров и без возвращаемого типа
void summ()
{
	int a, b;
	printf("enter a\n");
	scanf("%d", &a);
	printf("enter b\n");
	scanf("%d", &b);
	printf("%d + %d = %d", a, b, a+b);
}

//функция c параметрами и без возвращаемого типа
void summ(int a, int b)  //в данном случае можно использовать такое же название, т.к. функции отличаются количеством передаваемых параметров - перегрузка функций
{
	printf("%d + %d = %d", a, b, a+b);
}

int main()
{
	//вызов функции без параметров
	summ(); //при вызове любой функции необходимо указать её название и в () передать параметры, если параметров нет, то пустые скобки
	int a, b;
	printf("enter a\n");
	scanf("%d", &a);
	printf("enter b\n");
	scanf("%d", &b);
	//вызов функции с параметрами
	summ(a, b); //при передаче переменных в функцию типы данных указывать не нужно, мы передаем значения
	summ(2, 3); //можно сразу передать два числа
	summ(a, 10); //можно переменную и число
	return 0;
}

Перегрузка функций - это несколько функций в одной программе с одинаковым названием, но разным количеством или типом параметров.

Вызов функции без возвращаемого типа: 

						
имя_функции(передаваемые_в_функцию_параметры); //если параметры есть
имя_функции(); //если параметров нет

Функции с возвращаемым значением и без него.

Функции описанные в предыдущем примере, не очень удачные, т.к. функции ничего не возвращают (void), то результат работы функции мы никак не можем использовать в дальнейшем. Перепишем наши функции так, чтобы они возвращали результат.

						
#define _CRT_SECURE_NO_WARNINGS
#include <stdio.h>

//функция без параметров с возвращаемым типом int
int summ()  //возвращаемый тип int
{
	int a, b;
	printf("enter a\n");
	scanf("%d", &a);
	printf("enter b\n");
	scanf("%d", &b);
	return a + b; //возврат значения из функции
}

//функция c параметрами и возвращаемым типом int
int summ(int a, int b)  //в данном случае можно перегрузка функции
{
	return a + b; //возврат значения из функции
}

//функция c параметрами и возвращаемым типом double
double summ(double a, double b)  //в данном случае можно перегрузка функции по типам
{
	return a + b; //возврат значения из функции
}


int main()
{
	//вызов функции без параметров и с возвращаемым типом int
	int s = summ(); //т.к. функция возвращает значение, то его нужно сохранить в переменную
	printf("summa = %d", s);
	printf("summa = %d", summ());  //вызов функции summ при вызове функции printf
	int a, b;
	printf("enter a\n");
	scanf("%d", &a);
	printf("enter b\n");
	scanf("%d", &b);
	//вызов функции с параметрами и возвращаемым типом int
	s = summ(a, b);
	printf("summa = %d", s);
	s = summ( summ(1, 2), summ(3, 4)); //вызов функции как параметр для другой функции будет работать, получим 1+2+3+4
	printf("summa = %d", s);
	printf("summa = %lf", summ(3.5, 4.2)); //т.к. при вызове переданы числа типа double, то вызовется третий вариант функции summa
	return 0;
}

	#define _CRT_SECURE_NO_WARNINGS
	#include <stdio.h>

	float divide(int num, int divider)
	{
		if(divider!=0)
		{
			return (float)num/divider;
		}
		else
		{
			printf("Error! Division by zero!\n");
			return 0;
		}
	}

	void main()
	{
		float num = divide(10,0);
		printf("%f\n",num);
		num = divide(10,5);
		printf("%f\n",num);
	}

	#define _CRT_SECURE_NO_WARNINGS
	#include <stdio.h>

	int power(int x, int n)
	{
		int p=1;
		for(int i=0; i<n; i++)
		{
			p=p*x;
		}
		return p;
	}

	void main()
	{
		printf("2^5 = %d\n",power(2,5));
	}

		for(int i=2; i<x; i++)

			}
