Важность проблемы сортировок.

При хранении больших объемов данных возникает проблема поиска нужной информации. Для облегчения поиска данные необходимо сортировать.

Алгоритм сортировки — это алгоритм для упорядочивания элементов в списке(массиве). В случае, когда элемент списка имеет несколько полей, поле, служащее критерием порядка, называется ключом сортировки. На практике в качестве ключа часто выступает число, а в остальных полях хранятся какие-либо данные, никак не влияющие на работу алгоритма.

В зависимости от количества элементов массива, изменяется и время, необходимое на сортировку массива - поэтому и существует большое количество разных алгоритмов, которые отличаются друг от друга эффективностью и скоростью. Самый базовый учебный алгоритм - сортировка пузырьком. Он редко применяется в реальной жизни, но знать его наизусть - необходимо. Это простой, но достаточно эффективный алгоритм позволяющий понять принцип работы сортировки.

Пузырьковый алгоритм - подробный разбор.

Основная идея пузырьковой сортировки заключается в попарном сравнении двух соседних элементов массива. Если следующий элемент оказался меньше предыдущего, то меняем элементы местами. Таким образом после первого прохода массива самый большой элемент окажется на своем месте - в конце массива. Следующий раз в сортировке последний элемент уже сортировать не нужно, он и так окажется на своем месте. Повторив эту процедуру n-1 раз мы получим полностью отсортированный массив.

Рассмотрим алгоритм: 

						
#define _CRT_SECURE_NO_WARNINGS
#include <stdio.h>
#include <stdlib.h> 
#include <time.h>
int main()
{
	srand(time(NULL));
	const int n = 10;
	int mas[n];
	//заполняем наш массив случайными значениями
	for(int i=0; i<n; i++)
	{
		mas[i] = rand()%100;
		printf("%d\t",mas[i]); //закомментировать, если большой массив
	}
	printf("\n");
	//прогоняем массив n-1 раз, т.к. первый элемент окажется на своем месте автоматически, 
	//после упорядочивания всех остальных (n-1) элементов.
	for(int i=0; i<n; i++) 
	{
		//начинаем с начала и до конца. Отнимаем от n единицу, чтобы не выйти за пределы массива 
		//отнимаем i, так как "тяжелые" (утонувшие) элементы проверять уже не надо.
		for(int j=0; j<n-1-i; j++) 
		{
			if(mas[j] > mas[j+1]) //сравниваем два соседних элемента
			{
				//меняем элементы местами
				int buf = mas[j];
				mas[j] = mas[j+1];
				mas[j+1] = buf;
			}
		}
	}
	//выводим отсортированный массив
	printf("----------------------\n");
	for(int i=0; i<n; i++)
	{
		printf("%d\t",mas[i]);
	}
	return 0;
}

  • Создать массив, заполнить случайными числами, отсортировать половину массива по возрастанию, а вторую половину по убыванию. 

    • 						
#define _CRT_SECURE_NO_WARNINGS
#include <stdio.h>
#include <stdlib.h> 
#include <time.h>
int main()
{
	srand(time(NULL));
	const int n = 10;
	int mas[n];
	//заполняем наш массив случайными значениями
	for (int i = 0; i < n; i++)
	{
		mas[i] = rand() % 100;
		printf("%d\t", mas[i]); //закомментировать, если большой массив
	}
	printf("\n");
	for (int i = 0; i < n / 2; i++) //сортируем первую половину массива
	{
		for (int j = 0; j < n / 2 - 1 - i; j++)
		{
			if (mas[j] > mas[j + 1])
			{
				//меняем элементы местами
				int buf = mas[j];
				mas[j] = mas[j + 1];
				mas[j + 1] = buf;
			}
		}
	}
	for (int i = n / 2; i < n; i++) //сортируем вторую половину массива
	{
		for (int j = n / 2; j < n - 1 - (i - n / 2); j++)
		{
			if (mas[j] < mas[j + 1])
			{
				//меняем элементы местами
				int buf = mas[j];
				mas[j] = mas[j + 1];
				mas[j + 1] = buf;
			}
		}
	}
		
	printf("----------------------\n");
	for (int i = 0; i < n; i++)
	{
		printf("%d\t", mas[i]);
	}
	return 0;
}
