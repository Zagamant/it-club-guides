Введение в строки.

Строка — это массив символов char.
При работе со строками есть некоторые особенности. При использовании числовых массивов в размере массива обычно указывается реальное количество элементов. Т.е. объявляя int mas[10] мы делаем массив на 10 элементов и все 10 используем. При объявлении строки обычно указывается размер с запасом, т.к. мы не знаем какой длины будет введена строка.

Пример объявления: 

						
char str1[100];		//объявлена строка str1 с максимальным количеством элементов 100
char str2[200];		//объявлена строка str2 с максимальным количеством элементов 200

Никаких проблемм с избыточной длинной строки не будет, для обозначения длинна строки есть специальный символ '\0' и элементы записанные после этого символа выводиться не будут(там просто хранится какой-то мусор, но он нас ничем не мешает).

Рекомендуется все строки при объявлении сразу сделать пустой, т.е. записать в массив символ конца строки.

						
char str[100]={'\0'};  //пустая строка, в неё можно записать максимум 100 элементов(точнее 99 и символ конца строки)

Ввод и вывод строк. Проблема ввода строк после scanf (использование fflush).

Для ввода строки используется функция gets, для вывода - puts. Эти функции находится в библиотеке stdio.h.

Введем и выведем строку: 

						
char str[100] = {'\0'};
puts("Enter str:");
gets(str); //ввод строки
puts(str); //вывод строки

Если перед вводом строк использовалась функция scanf для ввода каких-либо переменных, то функция gets работать не будет, т.к. в потоке ввода остались старые значения и его необходимо очистить функцией fflush.

						
char str[100] = {'\0'};
puts("Enter str:");
fflush(stdin); //очистка буфера ввода, пишем, если перед вводом строки вводили что-то ещё
gets(str); //ввод строки
puts(str); //вывод строки

Кроме функции fflush можно воспользоваться таким способом:

						
char str[100] = {'\0'};
puts("Enter str:");
while (getchar() != '\n'); //альтернатива fflush
gets(str); //ввод строки
puts(str); //вывод строки

Нахождение конца строки и изменение элементов.

Зная, что каждая строка заканчивается символом конца строки '\0', мы можем легко перебрать все элементы строки и определить длину строки.

						
#define _CRT_SECURE_NO_WARNINGS
#include <stdio.h>
#include <locale.h>
int main()
{
	setlocale(LC_ALL, "Russian");
	char str[100] = {'\0'};
	puts("Введите строку:");
	gets(str); 
	int length=0;
	for(int i =0; str[i]!='\0'; i++)
	{
		length++;
	}	
	printf("длина строки: %d", length);
	return 0;
}

Заменим маленькие буквы 'a' в строке на большие буквы'A' и посчитаем их количество.

						
#define _CRT_SECURE_NO_WARNINGS
#include <stdio.h>
#include <locale.h>
int main()
{
	setlocale(LC_ALL, "Russian");
	char str[100] = {'\0'};
	puts("Введите строку:");
	gets(str); 
	int count=0;
	for(int i =0; str[i]!='\0'; i++)
	{
		if(str[i] == 'a')
		{
			str[i] = 'A';//замена маленькой буквы на большую
			count++;
		}
	}	
	puts(str);
	printf("количество букв а: %d", count);
	return 0;
}

#define _CRT_SECURE_NO_WARNINGS
#include <stdio.h>
#include <locale.h>
int main()
{
	setlocale(LC_ALL, "Russian");
	char str[100] = {'\0'};
	puts("Введите строку:");
	gets(str); 
	int count=1;
	for(int i =0; str[i]!='\0'; i++)
	{
		if(str[i] == ' ')
		{
			count++;
		}
	}	
	puts(str);
	printf("количество слов в строке: %d", count);
	return 0;
}
