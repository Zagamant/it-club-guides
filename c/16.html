Встроенные функции библиотеки string.h.

Полезные функции для работы со строками находятся в библиотеке string.h.

В таблице приведены только основные функции, которые чаще всего используются.

Название Описание
int strlen(char* str) опеределяет длину строки до нулевого символа('\0' не учитывается)
void strcpy(char* str2,char* str1); функция копирует все символы из строки str1 в строку str2
void strncpy(char* str2,char* str1, int n); функция копирует первых n символов из строки str1 в строку str2
void strcat(char* str2,char* str1); функция добавляет все символы из строки str1 в строку str2(конкатенация)
void strncat(char* str2,char* str1, int n); функция добавляет первых n символов из строки str1 в строку str2
int strcmp(char* str2,char* str1); функция сравнивает строку str1 и строку str2 с учетом регистра,
функция возвращает 0, если строки равны, 1 - первая больше и -1 - вторая больше
int strncmp(char* str2,char* str1); функция сравнивает первых n символов из строки str1 и строки str2 с учетом регистра,
функция возвращает 0, если строки равны, 1 - первая больше и -1 - вторая больше
int strcmpi(char* str2,char* str1); функция сравнивает строку str1 и строку str2 без учета регистра,
функция возвращает 0, если строки равны, 1 - первая больше и -1 - вторая больше

  • Программа демонстрирует работу различных функций из библиотеки string.h

	#define _CRT_SECURE_NO_WARNINGS
	#include <stdio.h>
	#include <string.h>
	int main()
	{
		setlocale(LC_ALL, "Russian");
		char str[100]={'\0'};
		gets(str);
		printf("Количество символов в строке - %d\n",strlen(str));
		char str2[100]={'\0'}, str3[100]={'\0'};
		gets(str2);
		strcpy(str3,str2); //копируем КУДА, ОТКУДА
		puts(str3);
		//функция strncpy отличается тем, что можно указать количество копируемых символов
		char str4[100] = {'\0'};
		strncpy(str4,str2,5);  //копируем 5 символов
		puts(str4);
		//строка, в которую копируются другие строки должна быть такой длины, чтобы все уместилось!
		char str5[100]={'\0'}, str6[100]={'\0'};
		gets(str5);
		gets(str6);
		strcat(str5, str6); //добавляем строку 6 к строке 5. Операция слепления строк называется КОНКАТЕНАЦИЕЙ (concatenation)
		puts(str5);
		strncat(str5,str6,2); //то же самое, но с ограничением длины
		puts(str5);
		strcat(str5, " TEXT"); //тоже работает
		puts(str5);
		char str7[100]={'\0'}, str8[100]={'\0'};
		gets(str7);
		gets(str8);
		//сравнение с учетом регистра
		//функция возвращает 0 если строки равны, 1 - если первая больше и -1 если первая строка меньше. 
		//сравнивается не длина, а коды символов
		if(strcmp(str7,str8)==0) 
		{
			printf("Строки равны\n");
		}
		//сравнение с учетом регистра и определенным количеством символов
		if(strncmp(str7,str8,5)==0) 
		{
			printf("Строки равны 2 \n");
		}
		//сравнение БЕЗ учета регистра!
		if(_strcmpi(str7,str8)==0)
		{
			printf("Строки равны 3 \n");
		}	
	}
